#include <stdio.h>

// 학생 정보를 저장할 구조체 정의
struct Student {
    char name[20];
    int kor;
    int eng;
    double avg; // 학생별 평균
};

int main() {
    // 1. 2명의 이름과 성적을 구조체 배열로 초기화
    struct Student students[2] = {
        {"홍길동", 90, 85, 0.0},
        {"이순신", 95, 92, 0.0}
    };

    // 2. 학생별 평균 계산
    for (int i = 0; i < 2; i++) {
        students[i].avg = (students[i].kor + students[i].eng) / 2.0;
    }

    // 3. 과목별 평균 계산
    double kor_total_avg = (students[0].kor + students[1].kor) / 2.0;
    double eng_total_avg = (students[0].eng + students[1].eng) / 2.0;

    // 4. 화면 출력
    printf("=== 학생별 성적 및 평균 ===\n");
    for (int i = 0; i < 2; i++) {
        printf("이름: %s | 국어: %d | 영어: %d | 평균: %.2f\n", 
               students[i].name, students[i].kor, students[i].eng, students[i].avg);
    }

    printf("\n=== 과목별 전체 평균 ===\n");
    printf("국어 과목 평균: %.2f\n", kor_total_avg);
    printf("영어 과목 평균: %.2f\n", eng_total_avg);

    return 0;
}
