#include <stdio.h>
#include <sqlite3.h>

// SELECT 결과를 처리할 콜백 함수
int callback(void* NotUsed, int argc, char** argv, char** azColName) {
    for (int i = 0; i < argc; i++) {
        printf("%s = %s\n", azColName[i], argv[i] ? argv[i] : "NULL");
    }
    printf("---------------------------\n");
    return 0;
}

int main() {
    sqlite3* db;
    char* err_msg = 0;
    int rc;

    // 1. 데이터베이스 파일 열기 (메모리 DB를 원하시면 "test.db" 대신 ":memory:" 사용 가능)
    rc = sqlite3_open("test.db", &db);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "데이터베이스를 열 수 없습니다: %s\n", sqlite3_errmsg(db));
        return 1;
    }
    printf("데이터베이스 연결 성공!\n\n");

    // 2. 테이블 생성 SQL문 실행
    char* sql_create = "DROP TABLE IF EXISTS Users;"
                       "CREATE TABLE Users(Id INTEGER PRIMARY KEY AUTOINCREMENT, Name TEXT, Age INTEGER);";
    
    rc = sqlite3_exec(db, sql_create, 0, 0, &err_msg);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "SQL 오류 (테이블 생성): %s\n", err_msg);
        sqlite3_free(err_msg);
        sqlite3_close(db);
        return 1;
    }

    // 3. 2명의 이름과 나이 생성 (데이터 삽입)
    char* sql_insert = "INSERT INTO Users (Name, Age) VALUES ('홍길동', 20);"
                       "INSERT INTO Users (Name, Age) VALUES ('이순신', 35);";

    rc = sqlite3_exec(db, sql_insert, 0, 0, &err_msg);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "SQL 오류 (데이터 삽입): %s\n", err_msg);
        sqlite3_free(err_msg);
        sqlite3_close(db);
        return 1;
    }
    printf("2명의 데이터 생성 완료.\n\n");

    // 4. DB에서 데이터 읽고 화면에 출력 (조회)
    char* sql_select = "SELECT Name, Age FROM Users;";
    printf("=== DB 데이터 출력 결과 ===\n");
    
    // callback 함수를 통해 조회된 레코드가 한 줄씩 출력됩니다.
    rc = sqlite3_exec(db, sql_select, callback, 0, &err_msg);
    if (rc != SQLITE_OK) {
        fprintf(stderr, "SQL 오류 (데이터 조회): %s\n", err_msg);
        sqlite3_free(err_msg);
        sqlite3_close(db);
        return 1;
    }

    // 5. 자원 해제 및 DB 닫기
    sqlite3_close(db);
    return 0;
}
