raylib를 nuget에서 설치한다.
```
#include "raylib.h"

typedef struct {
    float x;
    float y;
    float width;
    float height;
    int active;
    Color color;
} Block;

typedef struct {
    float x;
    float y;
    float vx;
    float vy;
    float radius;
} Ball;

typedef struct {
    float x;
    float y;
    float width;
    float height;
} Paddle;

typedef struct {
    int score;
    int failures;
    int blocks_left;
} GameState;

int main(void) {
    InitWindow(800, 600, "Block Breaker Game");
    SetTargetFPS(60);
    
    Block blocks[4][10];
    Ball ball;
    Paddle paddle;
    GameState state;
    int i, j;
    int game_running;
    int game_over;
    
    game_running = 1;
    game_over = 0;
    
    while (!WindowShouldClose() && game_running) {
        state.score = 0;
        state.failures = 0;
        state.blocks_left = 40;
        
        ball.x = 400.0f;
        ball.y = 550.0f;
        ball.vx = -3.0f;
        ball.vy = -5.0f;
        ball.radius = 5.0f;
        
        paddle.x = 350.0f;
        paddle.y = 560.0f;
        paddle.width = 100.0f;
        paddle.height = 15.0f;
        
        for (i = 0; i < 4; i++) {
            for (j = 0; j < 10; j++) {
                blocks[i][j].x = (float)(j * 80 + 20);
                blocks[i][j].y = (float)(i * 20 + 50);
                blocks[i][j].width = 75.0f;
                blocks[i][j].height = 17.0f;
                blocks[i][j].active = 1;
                
                if (i == 0) blocks[i][j].color = RED;
                else if (i == 1) blocks[i][j].color = ORANGE;
                else if (i == 2) blocks[i][j].color = YELLOW;
                else blocks[i][j].color = GREEN;
            }
        }
        
        game_over = 0;
        
        while (!WindowShouldClose() && !game_over) {
            if (IsKeyDown(KEY_LEFT) && paddle.x > 0.0f) {
                paddle.x -= 7.0f;
            }
            if (IsKeyDown(KEY_RIGHT) && paddle.x < 700.0f) {
                paddle.x += 7.0f;
            }
            
            ball.x += ball.vx;
            ball.y += ball.vy;
            
            if (ball.x - ball.radius < 0.0f || ball.x + ball.radius > 800.0f) {
                ball.vx = -(ball.vx);
            }
            if (ball.y - ball.radius < 0.0f) {
                ball.vy = -(ball.vy);
            }
            
            if (CheckCollisionCircleRec(
                (Vector2){ball.x, ball.y},
                ball.radius,
                (Rectangle){paddle.x, paddle.y, paddle.width, paddle.height}
            )) {
                ball.vy = -(ball.vy);
                ball.y = paddle.y - ball.radius;
                
                float paddle_center = paddle.x + paddle.width / 2.0f;
                float diff = ball.x - paddle_center;
                ball.vx = (diff / (paddle.width / 2.0f)) * 5.0f;
            }
            
            for (i = 0; i < 4; i++) {
                for (j = 0; j < 10; j++) {
                    if (!blocks[i][j].active) continue;
                    
                    if (CheckCollisionCircleRec(
                        (Vector2){ball.x, ball.y},
                        ball.radius,
                        (Rectangle){blocks[i][j].x, blocks[i][j].y, blocks[i][j].width, blocks[i][j].height}
                    )) {
                        blocks[i][j].active = 0;
                        ball.vy = -(ball.vy);
                        state.score += 10;
                        state.blocks_left -= 1;
                    }
                }
            }
            
            if (ball.y > 600.0f) {
                state.failures += 1;
                ball.x = 400.0f;
                ball.y = 550.0f;
                ball.vx = -3.0f;
                ball.vy = -5.0f;
            }
            
            if (state.failures >= 5) {
                game_over = 1;
            }
            
            if (state.blocks_left <= 0) {
                game_over = 1;
            }
            
            BeginDrawing();
            ClearBackground(BLACK);
            
            for (i = 0; i < 4; i++) {
                for (j = 0; j < 10; j++) {
                    if (blocks[i][j].active) {
                        DrawRectangle(
                            (int)blocks[i][j].x,
                            (int)blocks[i][j].y,
                            (int)blocks[i][j].width,
                            (int)blocks[i][j].height,
                            blocks[i][j].color
                        );
                        DrawRectangleLines(
                            (int)blocks[i][j].x,
                            (int)blocks[i][j].y,
                            (int)blocks[i][j].width,
                            (int)blocks[i][j].height,
                            DARKGRAY
                        );
                    }
                }
            }
            
            DrawCircle((int)ball.x, (int)ball.y, (int)ball.radius, WHITE);
            
            DrawRectangle((int)paddle.x, (int)paddle.y, (int)paddle.width, (int)paddle.height, SKYBLUE);
            DrawRectangleLines((int)paddle.x, (int)paddle.y, (int)paddle.width, (int)paddle.height, WHITE);
            
            DrawText(TextFormat("Score: %d", state.score), 20, 10, 20, WHITE);
            DrawText(TextFormat("Failures: %d/5", state.failures), 650, 10, 20, WHITE);
            DrawText(TextFormat("Blocks: %d", state.blocks_left), 350, 10, 20, WHITE);
            
            DrawText("LEFT/RIGHT: Move Paddle", 20, 570, 12, GRAY);
            
            EndDrawing();
        }
        
        BeginDrawing();
        ClearBackground(BLACK);
        
        if (state.failures >= 5) {
            DrawText("GAME OVER!", 300, 250, 50, RED);
            DrawText(TextFormat("Final Score: %d", state.score), 250, 350, 30, WHITE);
            DrawText("Press R to Restart or ESC to Exit", 200, 420, 20, GRAY);
        } else {
            DrawText("YOU WIN!", 300, 250, 50, GREEN);
            DrawText(TextFormat("Final Score: %d", state.score), 250, 350, 30, WHITE);
            DrawText("Press R to Restart or ESC to Exit", 200, 420, 20, GRAY);
        }
        
        EndDrawing();
        
        if (IsKeyPressed(KEY_R)) {
            continue;
        } else if (IsKeyPressed(KEY_ESCAPE)) {
            game_running = 0;
        }
    }
    ```
